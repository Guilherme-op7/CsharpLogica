public static void Exercicio12AD06()
{
    Console.WriteLine("Informe a altura do tri√¢ngulo:");
    int altura = Convert.ToInt32(Console.ReadLine());

    for (int i = 1; i <= altura; i++)
    {
       
        for (int j = altura - i; j > 0; j--)
        {
            Console.Write("  "); 
        }

        for (int k = 1; k <= i; k++)
        {
            Console.Write("* ");
        }

        Console.WriteLine();
    }
}
