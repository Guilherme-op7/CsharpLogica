    public static void Exercicio01AD05()
    {
        Console.WriteLine("Informe o primeiro lado do triângulo:");
        double lado1 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe o segundo lado do triângulo:");
        double lado2 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe o terceiro lado do triângulo:");
        double lado3 = Convert.ToDouble(Console.ReadLine());

        string tipo = VerificarTriangulo(lado1, lado2, lado3);

        Console.WriteLine($"Esse é um triângulo {tipo}");
    }

    public static string VerificarTriangulo(double lado1, double lado2, double lado3)
    {
        string tipo;

        if (lado1 == lado2 && lado2 == lado3)
        {
            tipo = "Equilátero";
        }

        else if (lado1 == lado2 || lado1 == lado3 || lado2 == lado3)
        {
            tipo = "Isósceles";
        }

        else
        {
            tipo = "Escaleno";
        }

        return tipo;
    }
