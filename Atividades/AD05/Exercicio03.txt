public static void Exercicio03AD05()
    {
        Console.WriteLine("Informe a primeira nota:");
        double nota1 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a segunda nota:");
        double nota2 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a terceira nota:");
        double nota3 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a quantidade de faltas:");
        int faltas = Convert.ToInt32(Console.ReadLine());

        double media = Media(nota1, nota2, nota3);


        string resp = Verificar(faltas, media);

        Console.WriteLine($"Situação do Aluno: {resp}");
    }

    public static double Media(double nota1, double nota2, double nota3)
    {
        double media = (nota1 + nota2 + nota3) / 3;

        return media;
    }

    public static string Verificar(int faltas, double media)
    {
        string situacao;

        if (faltas > 30)
        {
            situacao = "Reprovado";
        }

        else
        {
            if (media >= 8)
                situacao = "Aprovado com Sucesso";

            else if (media >= 6 && media < 8)
                situacao = "Aprovado";

            else if (media >= 3 && media < 6)
                situacao = "Recuperação";

            else if (media < 3 && media > 0)
                situacao = "Reprovado";

            else if (media == 0)
                situacao = "Desistente";

            else
                situacao = "Situação indefinida";
        }

        return situacao;
    }
