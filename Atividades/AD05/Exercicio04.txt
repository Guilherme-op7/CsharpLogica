    public static void Exercicio04AD05()
    {
        Console.WriteLine("Informe o primeiro valor:");
        double valor1 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe o segundo valor:");
        double valor2 = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe a operação (Soma, Subtração, Multiplicação, Divisão, Resto da Divisão, Potência):");
        string operacao = Console.ReadLine();

        double resultado = Calculo(operacao, valor1, valor2);

        Console.WriteLine($"Resultado é {resultado}");
    }

    public static double Calculo(string operacao, double valor1, double valor2)
    {
        double resultado;

        if (operacao == "soma")
        {
            resultado = valor1 + valor2;
        }

        else if (operacao == "subtracao")
        {
            resultado = valor1 - valor2;
        }

        else if (operacao == "multiplicacao")
        {
            resultado = valor1 * valor2;
        }

        else if (operacao == "divisao")
        {
            resultado = valor1 / valor2;
        }

        else if (operacao == "resto")
        {
            resultado = valor1 % valor2;
        }

        else if (operacao == "potencia")
        {
            resultado = 1;
            
            for (int i = 0; i < (int)valor2; i++)
            {
                resultado *= valor1;
            }
        }

        else
        {
            Console.WriteLine("Operação não suportada");
            return 0;
        }

        return resultado;
    }
