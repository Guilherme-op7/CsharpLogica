 public static void Exercicio09AD04()
    {
        Console.WriteLine("-- EXERCÍCIO 09 --");
        Console.WriteLine("Peso (kg):");
        double peso = Convert.ToDouble(Console.ReadLine());
        Console.WriteLine("Altura (m):");
        double altura = Convert.ToDouble(Console.ReadLine());

        double imc = CalcularImc2(peso, altura);

        Console.WriteLine($"Seu imc é {imc}");

        RespostaImc(imc);

    }

    public static double CalcularImc2(double peso, double altura)
    {
        double imc = peso / (altura * altura);

        return imc;
    }

    public static void RespostaImc(double imc)
    {
        if (imc >= 40)
        {
            Console.WriteLine("Obesidade Grau III");
        }

        else if (imc >= 35)
        {
            Console.WriteLine("Obesidade Grau II");
        }

        else if (imc >= 30)
        {
            Console.WriteLine("Obesidade Grau I");
        }

        else if (imc >= 25)
        {
            Console.WriteLine("Sobrepeso");
        }

        else if (imc >= 18.5)
        {
            Console.WriteLine("Peso Normal");
        }

        else
        {
            Console.WriteLine("Abaixo do Peso");
        }

    }
