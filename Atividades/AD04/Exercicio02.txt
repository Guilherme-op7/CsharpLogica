public static void Exericio02AD04()
    {
        Console.WriteLine("-- EXERCÍCIO 02 --");
        Console.WriteLine("Cor do semáforo:");
        string cor = Console.ReadLine().ToLower();

        CorSemaforo(cor);
    }

    public static string CorSemaforo(string cor)
    {
        if (cor == "vermelho")
        {
            Console.WriteLine("Espere");
        }

        else if (cor == "verde")
        {
            Console.WriteLine("Atravesse");
        }

        else
        {
            Console.WriteLine("Farol Inoperante");
        }

        return cor;
    }
