public static void Exercicio03AD04()
    {
        Console.WriteLine("-- EXERCÍCIO 03 --");
        Console.WriteLine("Ganhos:");
        double ganhos = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Gastos:");
        double gastos = Convert.ToDouble(Console.ReadLine());

        GastosCalculos(ganhos, gastos);
    }

    public static void GastosCalculos(double ganhos, double gastos)
    {
        if (ganhos >= gastos)
        {
            Console.WriteLine("Você está dentro do orçamento!");
        }

        else
        {
            Console.WriteLine("Você está fora do orçamento! Não gaste mais!");
        }

    }
