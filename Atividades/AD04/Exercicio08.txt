public static void Exercicio08AD04()
    {
        Console.WriteLine("-- EXERCÍCIO 08 --");
        Console.WriteLine("Digite as três notas:");
        double n1 = Convert.ToDouble(Console.ReadLine());
        double n2 = Convert.ToDouble(Console.ReadLine());
        double n3 = Convert.ToDouble(Console.ReadLine());

        double Calculo = CalculoMedia(n1, n2, n3);

        RespostaMedia(Calculo);

    }

    public static double CalculoMedia(double n1, double n2, double n3)
    {
        double media = (n1 + n2 + n3) / 3;

        return media;
    }

    public static void RespostaMedia(double media)
    {
        if (media == 0)
        {
            Console.WriteLine("Desistente");
        }


        else if (media < 3)
        {
            Console.WriteLine("Reprovado");
        }

        else if (media < 6)
        {
            Console.WriteLine("Recuperação");
        }

        else if (media < 8)
        {
            Console.WriteLine("Aprovado");
        }

        else
        {
            Console.WriteLine("Aprovado com Sucesso");
        }
    }
