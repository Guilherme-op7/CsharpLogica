public static void Exercicio01AD04()
    {
        Console.WriteLine("-- EXERCÍCIO 01 --");
        Console.WriteLine("Tipo do ingresso (Meia/Inteira):");
        string tipo = Console.ReadLine();

        Console.WriteLine("Quantidade:");
        int qtd = Convert.ToInt32(Console.ReadLine());

        double precoUnitario = ValorIngresso(tipo);
        double total = precoUnitario * qtd;

        Console.WriteLine("O total a pagar é R$ " + total);
    }

    public static double ValorIngresso(string tipo)
    {
        if (tipo == "meia")
        {
            return 14.25;
        }
        else
        {
            return 28.50;
        }
    }
